CREATE DATABASE diet ;

USE diet ;

CREATE TABLE USER
(
	_id INT PRIMARY KEY AUTO_INCREMENT,
	id VARCHAR(32) PRIMARY KEY NOT NULL,
	password VARCHAR(32) NOT NULL
) ENGINE=INNODB ;

CREATE TABLE food
(
	id INT PRIMARY KEY AUTO_INCREMENT,
	food_name  VARCHAR(64) NOT NULL,
	food_category VARCHAR(32) NOT NULL,
	food_detail_category VARCHAR(32) NOT NULL,
	serving_size FLOAT NOT NULL,
	calorie FLOAT NOT NULL,
	protein FLOAT NOT NULL,
	fat FLOAT NOT NULL,
	carbohydrate FLOAT NOT NULL,
	sugar FLOAT NOT NULL,
	calcium FLOAT NOT NULL,
	phosphorus FLOAT NOT NULL,
	iron FLOAT NOT NULL,
	salt FLOAT NOT NULL,
	potassium FLOAT NOT NULL,
	vitaminA FLOAT NOT NULL,
	vitaminB1 FLOAT NOT NULL,
	vitaminB2 FLOAT NOT NULL,
	niacin FLOAT NOT NULL,
	vitaminC FLOAT NOT NULL,
	folic_acid FLOAT NOT NULL
) ENGINE=INNODB DEFAULT CHARSET=utf8;
